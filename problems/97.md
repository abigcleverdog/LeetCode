97. Interleaving String
```
Given s1, s2, s3, find whether s3 is formed by the interleaving of s1 and s2.

For example,
Given:
s1 = "aabcc",
s2 = "dbbca",

When s3 = "aadbbcbcac", return true.
When s3 = "aadbbbaccc", return false.
```

class Solution:
    def isInterleave(self, s1, s2, s3):
        """
        :type s1: str
        :type s2: str
        :type s3: str
        :rtype: bool
        """
        if not s1:
            return s2 == s3
        if not s2:
            return s1 == s3
        
        l1, l2, l3 = len(s1), len(s2), len(s3)
        if l1+l2 != l3:
            return False
        
        t = [0] * 26
        for c in s1:
            t[ord(c) - 97] += 1
        for c in s2:
            t[ord(c) - 97] += 1
        for c in s3:
            t[ord(c) - 97] -= 1
        for n in t:
            if n != 0:
                return False
        
        def check(s1, s2, s3):
            if not s1:
                return s2 == s3
            if not s2:
                return s1 == s3
            if s1[0] == s3[0] and s2[0] == s3[0]:
                return check(s1[1:], s2, s3[1:]) or check(s1, s2[1:], s3[1:])
            if s1[0] != s3[0] and s2[0] != s3[0]:
                return False
            if s1[0] == s3[0]:
                return check(s1[1:], s2, s3[1:])
            if s2[0] == s3[0]:
                return check(s1, s2[1:], s3[1:])
        
        return check(s1, s2, s3)

class Solution:
    def numTrees(self, n):
        """
        :type n: int
        :rtype: int
        """
        if n==0:
            return 0
        
        treeNum=[1]*(n+1)
        
        for i in range(1,n+1,1):# 1 to n
            if i==1:
                treeNum[i]=1
            elif i==2:
                treeNum[i]=2
            else:
                cnt=0
                for mid in range(i):# 0 to i-1
                    cnt+=(treeNum[mid]*treeNum[(i-1)-mid])
                treeNum[i]=cnt
                
        return treeNum[n]
