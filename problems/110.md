class Solution(object):
    def minDepth(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        self.md = sys.maxint
        if not root:
            return 0
        
        def h(n, cd):
            if not n:
                return cd
            if cd>self.md:
                return cd
            cdc = cd+1
            if not n.left and not n.right:
                self.md = min(self.md, cdc)
                return cdc            
            l = h(n.left, cdc)
            r = h(n.right, cdc)            
            return min(l,r)
        h(root, 0)
        return self.md

class Solution(object):
    def minDepth(self, root):
        """
        :type root: TreeNode
        :rtype: int
        """
        if not root:
            return 0
        q = [(root,1)]
        while q:
            node, lvl = q.pop(0)
            if not node.left and not node.right:
                return lvl
            if node.left:
                q.append((node.left, lvl+1))
            if node.right:
                q.append((node.right, lvl+1))
